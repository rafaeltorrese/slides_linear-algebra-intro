\begin{frame}{Transposición de Producto de Matrices}
  Existe una propiedad que nos dice: \[ (AB)^{T} = B^T A^T \]

  \begin{columns}
    \column{0.5\textwidth}
  \begin{align*}
    \bm{A} & =%
             \begin{pmatrix}
               7&  1\\ 
               2& -3\\ 
               4&  8\\
             \end{pmatrix}
    \\
    \bm{B} &=%
             \begin{pmatrix}
               1& 6 \\
               -2& 3 \\
             \end{pmatrix}
    \\
    \bm{AB}^T &= %
                \begin{pmatrix}
                  5 &  8& -12\\
                  45&   3&  48\\
                \end{pmatrix}
  \end{align*}
  
  \column{0.5\textwidth}
  \begin{align*}
    \bm{B}^T &=%
               \begin{pmatrix}
                 1& -2 \\
                 6& 3 \\
               \end{pmatrix}
    \\
       \bm{A}^T & =%
             \begin{pmatrix}
               7&  2& 4 \\ 
               1& -3& 8 \\ 
             \end{pmatrix}
    \\
    \bm{B}^T\bm{A}^T &= %
                       \begin{pmatrix}
                         1& -2 \\
                         6& 3 \\
                       \end{pmatrix}
    \begin{pmatrix}
      7&  2& 4 \\ 
      1& -3& 8 \\ 
    \end{pmatrix}
  \end{align*}
  \end{columns}
